# Getting Started

<Steps>
### Install

Install the `@poetry-galore/rich-text-editor` package.

<Callout type={"info"}>
  For TypeScript definitions, install `@poetry-galore/types`. Recommended for
  those using TypeScript.
</Callout>

<Tabs items={['npm']}>
  <Tabs.Tab>

```bash copy
npm install @poetry-galore/rich-text-editor @poetry-galore/types
```

  </Tabs.Tab>
</Tabs>
 
### Ready to Go

You can now import `RichTextEditor` and use it in your project.

Create a component for your editor:

```tsx copy filename="components/editor.tsx"
import { useEffect, useState } from "react";

import RichTextEditor from "@poetry-galore/rich-text-editor";
import type { CustomEditorState } from "@poetry-galore/types";

function Editor() {
  // State variable to store the editor content as JSON
  // so that it can be saved or used as you'd like.
  const [editorStateJSON, setEditorStateJSON] = useState<any>();

  useEffect(() => {
    console.log(editorStateJSON);
    // You can do other stuff here like saving the contents
  }, [editorStateJSON]);

  // To be called when an update happens in the editor
  function onChange(customEditorState: CustomEditorState) {
    setEditorStateJSON(customEditorState.toJSON());
  }

  return (
    <RichTextEditor
      placeholderText="Start writing ..."
      onEditorChange={[{ onChange, ignoreSelectionChange: true }]}
    />
  );
}

export default Editor;
```

Use the component in your App:

```tsx copy filename="App.tsx"
import Editor from "./components/editor";

function App() {
  return (
    <div>
      <h1>Custom Editor</h1>
      <Editor />
    </div>
  );
}

export default App;
```
